on_actions = {
	on_startup = {
        effect = {
			every_country = {
				set_technology = { bicycle_infantry = 1 popup = no }
			}
		}
	}
	on_monthly = {
		effect = {
			every_state = {
				limit = {
					AND = {
						NOT = {
							is_core_of = THIS.controller
						}
						compliance > 49
					}
				}
				add_core_of = THIS.controller
			}
		}
	}
}
