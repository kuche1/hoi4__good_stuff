add_namespace = func_give_offmap_factories

country_event = {
    id = func_give_offmap_factories.1
    title = "Offmap Factories"
    desc = "You have been compensated a number of offmap factories based on how small your economy is. It may take a couple of hours until the new factories are registered by the game."
    picture = GFX_report_event_generic_read_write # GFX_report_event_japan_army_mountainside

    is_triggered_only = yes

    immediate = {

        # add 1 civ to prevent the theoretical
        # possibility of a crash due to division
        # by 0 error
        add_offsite_building = {
            type = industrial_complex
            level = 1
        }

        if = {
            limit = {
                num_of_factories < 73
            }

            set_variable = { var = factory_boost_ratio value = 73 }
            subtract_from_variable = { var = factory_boost_ratio value = num_of_factories }
            divide_variable = { var = factory_boost_ratio value = 3 }
            divide_variable = { var = factory_boost_ratio value = num_of_factories }

            set_variable = { var = compensation_civs value = num_of_civilian_factories }
            multiply_variable = { var = compensation_civs value = factory_boost_ratio }

            set_variable = { var = compensation_mils value = num_of_military_factories }
            multiply_variable = { var = compensation_mils value = factory_boost_ratio }

            set_variable = { var = compensation_docs value = num_of_naval_factories }
            multiply_variable = { var = compensation_docs value = factory_boost_ratio }

            add_offsite_building = {
                type = industrial_complex
                level = compensation_civs
            }
            add_offsite_building = {
                type = arms_factory
                level = compensation_mils
            }
            add_offsite_building = {
                type = dockyard
                level = compensation_docs
            }
        }

    }

}
