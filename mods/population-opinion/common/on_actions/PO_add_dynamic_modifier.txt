on_actions = {

	on_startup = {
    # it seems that `on_startup` does not actually get executed once for each country
    # like `on_monthly`, so you have to manually loop trough them

		effect = {
			every_possible_country = {
				set_variable = {
					var = PO_effect
					value = 0.00
				}

				add_dynamic_modifier = { modifier = PO_modifier }
			}
		}
	}

	on_monthly = {
		effect = {

			set_variable = { var = PO_effect value = num_of_civilian_factories }

			subtract_from_variable = { var = PO_effect value = num_of_military_factories }
			subtract_from_variable = { var = PO_effect value = num_of_naval_factories }

			multiply_variable = { var = PO_effect value = 0.004 }
			# multiplying by 0.01 translates into 1% for each factory

		}
	}

}
